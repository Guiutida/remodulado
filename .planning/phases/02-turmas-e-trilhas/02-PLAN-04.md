# PLAN-04: Conectar turma.html ao Backend

## Goal
Substituir todo o conteúdo hardcoded de `pages/turma.html` por um layout dual-view (`#view-professor` / `#view-aluno`) e criar `assets/js/turma.js` com lógica completa para as duas visões, consumindo as APIs de PLAN-02. Adicionalmente, conectar `assets/js/aluno.js` ao endpoint `GET /api/aluno/turma` para exibir nome da turma no painel do aluno.

## Requirements Covered
- TURM-01 (formulário de criação de turma com código exibido ao professor)
- TURM-02 (formulário de entrada por código para o aluno)
- TURM-03 (lista de membros da turma para o professor)
- TURM-04 (botão "remover" por aluno na lista de membros)
- TURM-05 (aluno vê nome da turma, professor, código e colegas)

## Pre-conditions
- PLAN-02 concluído: todos os endpoints de turma funcionando (POST /api/turmas, GET /api/turmas, GET /api/turmas/:id/membros, DELETE /api/turmas/:id/membros/:alunoId, POST /api/turmas/entrar, GET /api/aluno/turma)
- `assets/js/aluno.js` carregado ANTES de `turma.js` no HTML — fornece `usuarioLogado()`, `tokenAtual()`, `mostrarAviso()` globais

### Tarefa 2 — Criar `assets/js/turma.js`

Criar o arquivo com toda a lógica de dual-view. O arquivo carrega após `aluno.js`, portanto `usuarioLogado()`, `tokenAtual()` e `mostrarAviso()` já estão disponíveis globalmente — **não redeclarar**.

**Estrutura do arquivo:**

```js
// assets/js/turma.js
// Lógica de turma para professor e aluno.
// Carregado após aluno.js — usa usuarioLogado(), tokenAtual(), mostrarAviso() globais.

// ─── Controle de estado ───────────────────────────────────────────────────────
let turmaAtivaId = null; // turma selecionada pelo professor para ver membros

// ─── Inicialização ────────────────────────────────────────────────────────────

function inicializar() {
    const usuario = usuarioLogado();
    if (!usuario) {
        window.location.href = "../pages/login.html";
        return;
    }
    if (usuario.perfil === "professor") {
        document.getElementById("view-professor").style.display = "";
        inicializarProfessor();
    } else {
        document.getElementById("view-aluno").style.display = "";
        inicializarAluno();
    }
}

// ─── VISÃO DO PROFESSOR ──────────────────────────────────────────────────────

function inicializarProfessor() {
    carregarTurmasProfessor();
    document.getElementById("form-criar-turma").addEventListener("submit", async (ev) => {
        ev.preventDefault();
        const nome = document.getElementById("input-nome-turma").value.trim();
        const disciplina = document.getElementById("input-disciplina-turma").value.trim();
        const erroEl = document.getElementById("erro-criar-turma");
        erroEl.style.display = "none";
        try {
            const turma = await criarTurma({ nome, disciplina });
            mostrarAviso(`Turma "${turma.nome}" criada. Código: ${turma.codigo}`);
            document.getElementById("form-criar-turma").reset();
            await carregarTurmasProfessor();
        } catch (err) {
            erroEl.textContent = err.message;
            erroEl.style.display = "";
        }
    });
}

async function criarTurma(dados) {
    const resposta = await fetch("/api/turmas", {
        method: "POST",
        headers: {
            "Content-Type": "application/json",
            "Authorization": `Bearer ${tokenAtual()}`
        },
        body: JSON.stringify(dados)
    });
    if (resposta.status === 401) { redirecionarLogin(); return; }
    const resultado = await resposta.json();
    if (!resposta.ok) throw new Error(resultado.message || "Não foi possível criar a turma.");
    return resultado.turma;
}

async function carregarTurmasProfessor() {
    try {
        const resposta = await fetch("/api/turmas", {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (resposta.ok) renderizarListaTurmas(resultado.turmas);
    } catch {
        mostrarAviso("Não foi possível carregar as turmas.");
    }
}

function renderizarListaTurmas(turmas) {
    const container = document.getElementById("lista-turmas");
    if (!turmas.length) {
        container.innerHTML = "<p>Você ainda não criou nenhuma turma.</p>";
        return;
    }
    container.innerHTML = turmas.map((t) => `
        <article class="cartao-turma" data-turma-id="${t.id}">
            <strong>${t.nome}</strong>
            <span>${t.disciplina}</span>
            <code>Código: ${t.codigo}</code>
            <span>${t.total_alunos} aluno(s)</span>
            <button type="button" class="botao-secundario" data-acao="ver-membros" data-id="${t.id}" data-nome="${t.nome}">
                Ver membros
            </button>
        </article>
    `).join("");
    container.querySelectorAll("[data-acao='ver-membros']").forEach((btn) => {
        btn.addEventListener("click", () => {
            turmaAtivaId = parseInt(btn.dataset.id, 10);
            document.getElementById("titulo-membros").textContent = `Membros — ${btn.dataset.nome}`;
            document.getElementById("secao-membros").style.display = "";
            carregarMembros(turmaAtivaId);
        });
    });
}

async function carregarMembros(turmaId) {
    try {
        const resposta = await fetch(`/api/turmas/${turmaId}/membros`, {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (resposta.ok) renderizarMembros(resultado.membros, turmaId);
    } catch {
        mostrarAviso("Não foi possível carregar os membros.");
    }
}

function renderizarMembros(membros, turmaId) {
    const lista = document.getElementById("lista-membros");
    const semMembros = document.getElementById("sem-membros");
    if (!membros.length) {
        lista.innerHTML = "";
        semMembros.style.display = "";
        return;
    }
    semMembros.style.display = "none";
    lista.innerHTML = membros.map((m) => `
        <li data-membro-id="${m.id}">
            <span>${m.nome}</span>
            <small>${m.email}</small>
            <button type="button" class="botao-perigo" data-acao="remover" data-turma="${turmaId}" data-aluno="${m.id}" data-nome="${m.nome}">
                Remover
            </button>
        </li>
    `).join("");
    lista.querySelectorAll("[data-acao='remover']").forEach((btn) => {
        btn.addEventListener("click", () => removerMembro(
            parseInt(btn.dataset.turma, 10),
            parseInt(btn.dataset.aluno, 10),
            btn.dataset.nome
        ));
    });
}

async function removerMembro(turmaId, alunoId, nomeAluno) {
    if (!confirm(`Remover "${nomeAluno}" da turma?`)) return;
    try {
        const resposta = await fetch(`/api/turmas/${turmaId}/membros/${alunoId}`, {
            method: "DELETE",
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (resposta.ok) {
            mostrarAviso(`${nomeAluno} removido da turma.`);
            carregarMembros(turmaId);
        } else {
            mostrarAviso(resultado.message || "Erro ao remover membro.");
        }
    } catch {
        mostrarAviso("Não foi possível remover o membro.");
    }
}

// ─── VISÃO DO ALUNO ──────────────────────────────────────────────────────────

function inicializarAluno() {
    carregarTurmaAluno();
    document.getElementById("form-entrar-turma").addEventListener("submit", async (ev) => {
        ev.preventDefault();
        const codigo = document.getElementById("input-codigo-turma").value.trim().toUpperCase();
        const erroEl = document.getElementById("erro-entrar-turma");
        erroEl.style.display = "none";
        try {
            const turma = await entrarNaTurma(codigo);
            mostrarAviso(`Bem-vindo à turma "${turma.nome}"!`);
            renderizarTurmaAluno(turma);
            document.getElementById("painel-entrar-turma").style.display = "none";
            document.getElementById("painel-info-turma").style.display = "";
        } catch (err) {
            erroEl.textContent = err.message;
            erroEl.style.display = "";
        }
    });
}

async function entrarNaTurma(codigo) {
    const resposta = await fetch("/api/turmas/entrar", {
        method: "POST",
        headers: {
            "Content-Type": "application/json",
            "Authorization": `Bearer ${tokenAtual()}`
        },
        body: JSON.stringify({ codigo })
    });
    if (resposta.status === 401) { redirecionarLogin(); return; }
    const resultado = await resposta.json();
    if (!resposta.ok) throw new Error(resultado.message || "Código inválido.");
    return resultado.turma;
}

async function carregarTurmaAluno() {
    try {
        const resposta = await fetch("/api/aluno/turma", {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        if (resposta.status === 404) {
            // Aluno não está em turma — mostrar formulário de entrada
            document.getElementById("painel-entrar-turma").style.display = "";
            document.getElementById("painel-info-turma").style.display = "none";
            return;
        }
        const resultado = await resposta.json();
        if (resposta.ok) {
            document.getElementById("painel-entrar-turma").style.display = "none";
            document.getElementById("painel-info-turma").style.display = "";
            renderizarTurmaAluno(resultado.turma);
        }
    } catch {
        mostrarAviso("Não foi possível carregar sua turma.");
    }
}

function renderizarTurmaAluno(turma) {
    document.getElementById("nome-turma-aluno").textContent = turma.nome || "—";
    document.getElementById("professor-turma").textContent = turma.professor_nome || "—";
    document.getElementById("disciplina-turma").textContent = turma.disciplina || "—";
    document.getElementById("codigo-turma").textContent = turma.codigo || "—";
    // total_membros inclui o próprio aluno; exibir colegas = total - 1
    const colegas = Math.max(0, (turma.total_membros || 1) - 1);
    document.getElementById("total-colegas").textContent = colegas;
}

// ─── Utilitários ─────────────────────────────────────────────────────────────

function redirecionarLogin() {
    localStorage.removeItem("duopratic_token");
    localStorage.removeItem("duopratic_usuario");
    window.location.href = "../pages/login.html";
}

// ─── Iniciar ──────────────────────────────────────────────────────────────────
inicializar();
```

## Threat Model

| Ameaça STRIDE | Componente | Disposição | Mitigação |
|---------------|------------|------------|-----------|
| Spoofing | `localStorage.duopratic_usuario.perfil` pode ser manipulado pelo usuário | aceitar (frontend-only) | Dual-view é apenas conveniência de UI; toda autorização real está no backend via JWT + guards de perfil; manipular o localStorage só muda o que o usuário vê, não o que a API aceita |
| Information Disclosure | Token JWT exposto em `localStorage` | aceitar | Padrão existente do projeto (Phase 1); alternativa (httpOnly cookie) é escopo de Phase 5 |
| Tampering | XSS via `innerHTML` com dados da API | mitigar | Conteúdo de `turma.nome`, `turma.codigo` inserido via `textContent` nos campos de info; `innerHTML` usado apenas em templates com campos sanitizados sem `<script>` — nomes de turma não executam HTML se usados em `innerHTML` no template de lista; adicionar `.replace(/</g, "&lt;")` se o projeto evoluir para aceitar HTML arbitrário |
| Repudiation | Remoção de membro sem confirmação | mitigar | `confirm()` antes de `DELETE` garante confirmação explícita do professor |
| Denial of Service | Clique repetido em "Criar turma" antes da resposta chegar | aceitar | Formulário reset após sucesso; back-end retorna turma criada; duplicata de nome não é proibida (apenas código é UNIQUE) — comportamento aceitável para TCC |
