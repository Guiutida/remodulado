# PLAN-05: Conectar trilhas.html ao Backend

## Goal
Corrigir o guard de `trilhas.html` de `"aluno"` para `""`, substituir todo o conteúdo hardcoded por um layout dual-view (`#view-professor` / `#view-aluno`) e criar `assets/js/trilhas.js` com lógica completa para professor criar trilhas/etapas e aluno navegar/marcar progresso, consumindo as APIs de PLAN-03.

## Requirements Covered
- TRIL-01 (professor cria trilha com nome, disciplina, descrição via formulário)
- TRIL-02 (professor adiciona etapas com tipo texto/vídeo/link via formulário)
- TRIL-03 (atribuição da trilha a uma turma selecionada no formulário)
- TRIL-04 (aluno vê lista de trilhas disponíveis na sua turma com percentual)
- TRIL-05 (aluno navega pelas etapas em sequência via `ordem ASC`)
- TRIL-06 (barra de progresso calculada pelo backend via SUM(IF()) — não hardcoded)
- TRIL-07 (botão "Marcar como concluída" por etapa persiste no banco)

## Pre-conditions
- PLAN-01 concluído: tabelas `trilhas`, `trilha_etapas`, `progresso_etapa` existem no banco
- PLAN-02 concluído: `GET /api/turmas` e `GET /api/aluno/turma` funcionando (necessários para os dropdowns)
- PLAN-03 concluído: todos os endpoints de trilhas funcionando
- `assets/js/aluno.js` carregado antes de `trilhas.js` — fornece `usuarioLogado()`, `tokenAtual()`, `mostrarAviso()` globais

### Tarefa 2 — Criar `assets/js/trilhas.js`

Criar o arquivo com toda a lógica dual-view. O arquivo carrega após `aluno.js` — `usuarioLogado()`, `tokenAtual()` e `mostrarAviso()` já existem globalmente, **não redeclarar**.

**Regra de seletores DOM:** usar `[data-trilha-id]`, `[data-etapa-id]`, `[data-progresso-trilha]` — nunca `[data-atividade-funcoes]`, `[data-status-funcoes]`, `.etapas-trilha` (usados por `aluno.js` — colisão proibida).

**Estrutura do arquivo:**

```js
// assets/js/trilhas.js
// Lógica de trilhas para professor e aluno.
// Carregado após aluno.js — usa usuarioLogado(), tokenAtual(), mostrarAviso() globais.

// ─── Estado ───────────────────────────────────────────────────────────────────
let trilhaAtiva = null; // { id, titulo } da trilha selecionada pelo professor
let trilhaSelecionadaAluno = null; // id da trilha que o aluno está navegando

// ─── Inicialização ────────────────────────────────────────────────────────────

function inicializar() {
    const usuario = usuarioLogado();
    if (!usuario) { redirecionarLogin(); return; }
    if (usuario.perfil === "professor") {
        document.getElementById("view-professor").style.display = "";
        inicializarProfessor();
    } else {
        document.getElementById("view-aluno").style.display = "";
        inicializarAluno();
    }
}

// ─── VISÃO DO PROFESSOR ──────────────────────────────────────────────────────

function inicializarProfessor() {
    carregarTurmasProfessor();
    document.getElementById("select-turma-prof").addEventListener("change", (ev) => {
        const turmaId = ev.target.value;
        trilhaAtiva = null;
        document.getElementById("secao-adicionar-etapa").style.display = "none";
        if (!turmaId) {
            document.getElementById("form-criar-trilha").style.display = "none";
            document.getElementById("lista-trilhas-prof").innerHTML = "";
            return;
        }
        document.getElementById("form-criar-trilha").style.display = "";
        document.getElementById("form-criar-trilha").dataset.turmaId = turmaId;
        carregarTrilhasProfessor(parseInt(turmaId, 10));
    });
    document.getElementById("form-criar-trilha").addEventListener("submit", async (ev) => {
        ev.preventDefault();
        const turmaId = parseInt(ev.target.dataset.turmaId, 10);
        const titulo = document.getElementById("input-titulo-trilha").value.trim();
        const disciplina = document.getElementById("input-disciplina-trilha").value.trim();
        const descricao = document.getElementById("input-descricao-trilha").value.trim();
        const erroEl = document.getElementById("erro-criar-trilha");
        erroEl.style.display = "none";
        try {
            await criarTrilha({ titulo, disciplina, descricao: descricao || undefined, turma_id: turmaId });
            mostrarAviso(`Trilha "${titulo}" criada com sucesso.`);
            ev.target.reset();
            ev.target.dataset.turmaId = turmaId; // preservar após reset
            await carregarTrilhasProfessor(turmaId);
        } catch (err) {
            erroEl.textContent = err.message;
            erroEl.style.display = "";
        }
    });
    document.getElementById("form-adicionar-etapa").addEventListener("submit", async (ev) => {
        ev.preventDefault();
        if (!trilhaAtiva) return;
        const titulo = document.getElementById("input-titulo-etapa").value.trim();
        const tipo = document.getElementById("input-tipo-etapa").value;
        const conteudo = document.getElementById("input-conteudo-etapa").value.trim();
        const erroEl = document.getElementById("erro-adicionar-etapa");
        erroEl.style.display = "none";
        try {
            await adicionarEtapa(trilhaAtiva.id, { titulo, tipo, conteudo });
            mostrarAviso(`Etapa "${titulo}" adicionada.`);
            ev.target.reset();
            await carregarEtapasProfessor(trilhaAtiva.id);
        } catch (err) {
            erroEl.textContent = err.message;
            erroEl.style.display = "";
        }
    });
}

async function carregarTurmasProfessor() {
    try {
        const resposta = await fetch("/api/turmas", {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (!resposta.ok) return;
        const select = document.getElementById("select-turma-prof");
        resultado.turmas.forEach((t) => {
            const opt = document.createElement("option");
            opt.value = t.id;
            opt.textContent = `${t.nome} (${t.disciplina})`;
            select.appendChild(opt);
        });
    } catch {
        mostrarAviso("Não foi possível carregar as turmas.");
    }
}

async function criarTrilha(dados) {
    const resposta = await fetch("/api/trilhas", {
        method: "POST",
        headers: {
            "Content-Type": "application/json",
            "Authorization": `Bearer ${tokenAtual()}`
        },
        body: JSON.stringify(dados)
    });
    if (resposta.status === 401) { redirecionarLogin(); return; }
    const resultado = await resposta.json();
    if (!resposta.ok) throw new Error(resultado.message || "Não foi possível criar a trilha.");
    return resultado.trilha;
}

async function carregarTrilhasProfessor(turmaId) {
    try {
        const resposta = await fetch(`/api/trilhas/turma/${turmaId}`, {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (resposta.ok) renderizarTrilhasProfessor(resultado.trilhas);
    } catch {
        mostrarAviso("Não foi possível carregar as trilhas.");
    }
}

function renderizarTrilhasProfessor(trilhas) {
    const container = document.getElementById("lista-trilhas-prof");
    if (!trilhas.length) {
        container.innerHTML = "<p>Nenhuma trilha nesta turma ainda.</p>";
        return;
    }
    container.innerHTML = trilhas.map((t) => `
        <article data-trilha-id="${t.id}">
            <strong>${t.titulo}</strong>
            <span>${t.disciplina}</span>
            <span>${t.total_etapas} etapa(s)</span>
            <button type="button" class="botao-secundario" data-acao="gerenciar-etapas" data-id="${t.id}" data-titulo="${t.titulo}">
                Gerenciar etapas
            </button>
        </article>
    `).join("");
    container.querySelectorAll("[data-acao='gerenciar-etapas']").forEach((btn) => {
        btn.addEventListener("click", () => {
            trilhaAtiva = { id: parseInt(btn.dataset.id, 10), titulo: btn.dataset.titulo };
            document.getElementById("titulo-etapas-prof").textContent = `Etapas — ${trilhaAtiva.titulo}`;
            document.getElementById("secao-adicionar-etapa").style.display = "";
            carregarEtapasProfessor(trilhaAtiva.id);
        });
    });
}

async function adicionarEtapa(trilhaId, dados) {
    const resposta = await fetch(`/api/trilhas/${trilhaId}/etapas`, {
        method: "POST",
        headers: {
            "Content-Type": "application/json",
            "Authorization": `Bearer ${tokenAtual()}`
        },
        body: JSON.stringify(dados)
    });
    if (resposta.status === 401) { redirecionarLogin(); return; }
    const resultado = await resposta.json();
    if (!resposta.ok) throw new Error(resultado.message || "Não foi possível adicionar a etapa.");
    return resultado.etapa;
}

async function carregarEtapasProfessor(trilhaId) {
    try {
        const resposta = await fetch(`/api/trilhas/${trilhaId}/etapas`, {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (!resposta.ok) return;
        const lista = document.getElementById("lista-etapas-prof");
        if (!resultado.etapas.length) {
            lista.innerHTML = "<li>Nenhuma etapa ainda.</li>";
            return;
        }
        lista.innerHTML = resultado.etapas.map((e) => `
            <li data-etapa-id="${e.id}">
                <strong>${e.ordem}. ${e.titulo}</strong>
                <span class="tipo-etapa">[${e.tipo}]</span>
            </li>
        `).join("");
    } catch {
        mostrarAviso("Não foi possível carregar as etapas.");
    }
}

// ─── VISÃO DO ALUNO ──────────────────────────────────────────────────────────

function inicializarAluno() {
    carregarTrilhasAluno();
    document.getElementById("btn-voltar-trilhas").addEventListener("click", () => {
        trilhaSelecionadaAluno = null;
        document.getElementById("painel-etapas-trilha").style.display = "none";
        document.getElementById("painel-lista-trilhas").style.display = "";
    });
}

async function carregarTrilhasAluno() {
    try {
        const resposta = await fetch("/api/trilhas/minhas", {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (resposta.ok) renderizarTrilhasAluno(resultado.trilhas);
    } catch {
        mostrarAviso("Não foi possível carregar as trilhas.");
    }
}

function renderizarTrilhasAluno(trilhas) {
    const container = document.getElementById("lista-trilhas-aluno");
    if (!trilhas.length) {
        container.innerHTML = "<p>Sua turma ainda não possui trilhas. Aguarde o professor adicionar.</p>";
        return;
    }
    container.innerHTML = trilhas.map((t) => `
        <article data-trilha-id="${t.id}">
            <strong>${t.titulo}</strong>
            <span>${t.disciplina}</span>
            <p>${t.descricao || ""}</p>
            <div class="barra-progresso-container">
                <div class="barra">
                    <span data-progresso-trilha="${t.id}" style="width:${t.percentual || 0}%"></span>
                </div>
                <span>${t.percentual || 0}%</span>
            </div>
            <small>${t.etapas_concluidas || 0} de ${t.total_etapas || 0} etapas concluídas</small>
            <button type="button" class="botao" data-acao="abrir-trilha" data-id="${t.id}" data-titulo="${t.titulo}">
                ${(t.percentual || 0) > 0 ? "Continuar" : "Iniciar"}
            </button>
        </article>
    `).join("");
    container.querySelectorAll("[data-acao='abrir-trilha']").forEach((btn) => {
        btn.addEventListener("click", () => abrirTrilha(
            parseInt(btn.dataset.id, 10),
            btn.dataset.titulo
        ));
    });
}

async function abrirTrilha(trilhaId, titulo) {
    trilhaSelecionadaAluno = trilhaId;
    document.getElementById("titulo-trilha-selecionada").textContent = titulo;
    document.getElementById("painel-lista-trilhas").style.display = "none";
    document.getElementById("painel-etapas-trilha").style.display = "";
    await carregarEtapasAluno(trilhaId);
    await atualizarProgressoVisual(trilhaId);
}

async function carregarEtapasAluno(trilhaId) {
    try {
        const resposta = await fetch(`/api/trilhas/${trilhaId}/etapas`, {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (resposta.ok) renderizarEtapasAluno(resultado.etapas, trilhaId);
    } catch {
        mostrarAviso("Não foi possível carregar as etapas.");
    }
}

function renderizarEtapasAluno(etapas, trilhaId) {
    const lista = document.getElementById("lista-etapas-aluno");
    if (!etapas.length) {
        lista.innerHTML = "<li>Esta trilha ainda não possui etapas.</li>";
        return;
    }
    lista.innerHTML = etapas.map((e) => {
        const concluido = e.concluido === true;
        const conteudoHtml = e.tipo === "video"
            ? `<a href="${e.conteudo}" target="_blank" rel="noopener">▶ Assistir vídeo</a>`
            : e.tipo === "link"
                ? `<a href="${e.conteudo}" target="_blank" rel="noopener">🔗 Acessar link</a>`
                : `<p class="conteudo-texto">${e.conteudo}</p>`;
        return `
            <li class="etapa-item ${concluido ? "etapa-concluida" : ""}" data-etapa-id="${e.id}">
                <h4>${e.ordem}. ${e.titulo}</h4>
                ${conteudoHtml}
                ${concluido
                    ? `<span class="badge-concluida">✓ Concluída</span>`
                    : `<button type="button" class="botao" data-acao="concluir-etapa" data-etapa="${e.id}" data-trilha="${trilhaId}">
                           Marcar como concluída
                       </button>`
                }
            </li>
        `;
    }).join("");
    lista.querySelectorAll("[data-acao='concluir-etapa']").forEach((btn) => {
        btn.addEventListener("click", () => concluirEtapa(
            parseInt(btn.dataset.etapa, 10),
            parseInt(btn.dataset.trilha, 10)
        ));
    });
}

async function concluirEtapa(etapaId, trilhaId) {
    try {
        const resposta = await fetch(`/api/trilhas/etapas/${etapaId}/concluir`, {
            method: "POST",
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (resposta.status === 401) { redirecionarLogin(); return; }
        const resultado = await resposta.json();
        if (!resposta.ok) {
            mostrarAviso(resultado.message || "Não foi possível salvar o progresso.");
            return;
        }
        mostrarAviso("Etapa marcada como concluída!");
        // Atualizar visualmente sem recarregar a página inteira
        await carregarEtapasAluno(trilhaId);
        await atualizarProgressoVisual(trilhaId);
    } catch {
        mostrarAviso("Erro ao salvar progresso. Tente novamente.");
    }
}

async function atualizarProgressoVisual(trilhaId) {
    try {
        const resposta = await fetch(`/api/trilhas/${trilhaId}/progresso`, {
            headers: { "Authorization": `Bearer ${tokenAtual()}` }
        });
        if (!resposta.ok) return;
        const resultado = await resposta.json();
        const pct = resultado.progresso?.percentual || 0;
        const fill = document.getElementById("fill-progresso-trilha");
        const texto = document.getElementById("texto-progresso-trilha");
        if (fill) fill.style.width = `${pct}%`;
        if (texto) texto.textContent = `${pct}%`;
    } catch { /* silencioso */ }
}

// ─── Utilitários ──────────────────────────────────────────────────────────────

function redirecionarLogin() {
    localStorage.removeItem("duopratic_token");
    localStorage.removeItem("duopratic_usuario");
    window.location.href = "../pages/login.html";
}

// ─── Iniciar ──────────────────────────────────────────────────────────────────
inicializar();
```

| Ameaça STRIDE | Componente | Disposição | Mitigação |
|---------------|------------|------------|-----------|
| Elevation of Privilege | Professor antes era redirecionado (guard "aluno"); após mudança para "", qualquer autenticado acessa | aceitar | Guard `""` é intencional — professores precisam criar trilhas; autorização granular está no backend (guards de perfil por endpoint) |
| Spoofing | `localStorage.perfil` manipulado para acessar view errada | aceitar | View é apenas UI; backend rejeita ações não autorizadas pelo perfil (professor tentando `GET /api/trilhas/minhas` recebe 403; aluno tentando `POST /api/trilhas` recebe 403) |
| Information Disclosure | `innerHTML` com dados de trilha/etapa pode conter HTML injetado | mitigar | Títulos e descrições inseridos em templates; conteúdo de `tipo=link` ou `tipo=video` renderizado como `href` — usuário só vê o link, não executa JS; para `tipo=texto`, `e.conteudo` inserido dentro de `<p>` via innerHTML sem executar scripts. Para produção, usar `textContent` no conteúdo de texto ou sanitizador |
| Tampering | Aluno manipula `data-etapa` no DOM para concluir etapa de outra turma | mitigar | `concluirEtapa` no backend verifica membership via JOIN antes do INSERT; manipulação do DOM não bypassa a verificação |
| Denial of Service | Clique rápido repetido em "Marcar como concluída" | aceitar | `ON DUPLICATE KEY UPDATE` é idempotente no backend; múltiplos cliques apenas atualizam `concluido_em`; sem side effects |
