module.exports = function errorHandler(erro, req, res, next) {
    if (erro.code === "ER_DUP_ENTRY") {
        return res.status(409).json({ status: "erro", message: "Este e-mail já está cadastrado." });
    }
    res.status(erro.status || 500).json({
        status: "erro",
        message: erro.message || "Erro interno do servidor.",
        ...(process.env.NODE_ENV !== "production" && { detalhe: erro.stack })
    });
};

| Express → Cliente | Respostas 404/500 não devem expor internals (stack trace, estrutura de diretórios) |

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-05-06 | Denial of Service | unhandledRejection sem handler | mitigate | `process.on('unhandledRejection')` com log + process.exit(1) — impede crash silencioso |
| T-05-07 | Information Disclosure | 404 com Express default | mitigate | handler404 retorna JSON sem revelar estrutura interna de rotas |
| T-05-08 | Information Disclosure | GEMINI_API_KEY/JWT_SECRET ausentes | mitigate | `validarEnv()` encerra o processo antes de qualquer request ser processado |
| T-05-09 | Tampering | helmet CSP img-src | mitigate | `'img-src': ["'self'", "data:"]` permite imagens de /uploads (same-origin) sem abrir cross-origin |
